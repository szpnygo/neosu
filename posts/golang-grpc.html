Golang

gRPC 是一个高性能、开源和通用的 RPC 框架，面向移动和 HTTP/2 设计。

所谓RPC(remote procedure call 远程过程调用)框架实际是提供了一套机制，使得应用程序之间可以进行通信，而且也遵从server/client模型。使用的时候客户端调用server端提供的接口就像是调用本地的函数一样。

安装grpc包

首先我们需要安装和准备grpc所需要的环境

go get google.golang.org/grpc

同时需要下载和安装protocol buffer编译器

https://github.com/protocolbuffers/protobuf/releases

这里我下载的是protoc-3.17.1-osx-x86_64.zip，解压放到环境变量中就可以直接使用。

export PROTOC="/Users/neosu/protoc-3.17.1-osx-x86_64/bin"
export PATH=$PROTOC:$PATH

定义服务

首先第一步是使用protocol buffer定义gRPC服务还有方法的请求和响应类型。 创建hello.proto文件

首先写入默认的一些东西

//版本号
syntax = "proto3";
//go的包
option go_package = "neobaran.com/study/golang/grpc/client/hello";
package hello;

然后定义服务

service HelloRouter {
    //这就是一个普通的请求
    rpc SayHello(HelloRequest) returns (HelloReply){}
    //这是一个流请求
    rpc GetStream (StreamRequestData) returns (stream StreamResponseData);
}

message HelloRequest{
    string name = 1;
}

message HelloReply{
    string message = 2;
}

message StreamRequestData{
    string data = 1;
}

message StreamResponseData{
    string data = 1;
}

生成客户端和服务端代码

protoc --go_out=plugins=grpc:./hello  hello/hello.proto

此命令会生成hello.pb.go文件，引入到我们的项目中。

服务端代码

我们开始创建我们的服务端代码

package main

import (
	"context"
	"fmt"
	"log"
	"net"
	"time"

	"google.golang.org/grpc"
	"neobaran.com/study/golang/grpc/server/hello"
)

//定义一个struct
type HelloServer struct{}

//实现我们在之前proto中定义的方法
func (s *HelloServer) SayHello(ctx context.Context, req *hello.HelloRequest) (*hello.HelloReply, error) {
	log.Println("request : " + req.Name)
	return &hello.HelloReply{Message: "hello world"}, nil
}

//一个流的方法
func (s *HelloServer) GetStream(req *hello.StreamRequestData, res hello.HelloRouter_GetStreamServer) error {
    //每秒自动输出时间
	for {
		time.Sleep(1 * time.Second)
		res.Send(&hello.StreamResponseData{Data: fmt.Sprintf("%v", time.Now().Unix())})
	}
}

func main() {
    //设置gRPC的端口监听
	lis, _ := net.Listen("tcp", ":8081")
    //开启一个gRPC服务
	s := grpc.NewServer()
    //注册我们的服务
	hello.RegisterHelloRouterServer(s, &HelloServer{})
	log.Println("grpc server run")
    //开启监听
	s.Serve(lis)
}

这样我们的服务端代码就写好了，运行起来，我们接下来去写客户端代码

客户端代码

package main

import (
	"context"
	"fmt"
	"log"

	"google.golang.org/grpc"
	"neobaran.com/study/golang/grpc/client/hello"
)

func main() {
    //发起链接
	conn, err := grpc.Dial("127.0.0.1:8081", grpc.WithInsecure())
	if err != nil {
		log.Fatalf("grpc did not connect : %v", err)
	}
	defer conn.Close()
    //开启gRPC客户端
	c := hello.NewHelloRouterClient(conn)

	name := "NeoSu"

    //调用SayHello方法
	r, err := c.SayHello(context.Background(), &hello.HelloRequest{Name: name})
	if err != nil {
		log.Fatalf("request error : %v", err)
	}
	log.Println(r.Message)

    //请求我们定义的那个流的方法
	res, _ := c.GetStream(context.Background(), &hello.StreamRequestData{Data: "test"})
	result, _ := res.Recv()
	fmt.Println("result:", result.Data)
	for {
        //不断的接受流内容
		data, recvErr := res.Recv()
		if recvErr != nil {
			log.Println(recvErr.Error())
			break
		}
		log.Println(data.Data)
	}

}
