322.零钱兑换

2021-05-13
2分钟阅读时长
算法 , LeetCode
动态规划

题目

给定不同面额的硬币 coins 和一个总金额 amount。编写一个函数来计算可以凑成总金额所需的最少的硬币个数。如果没有任何一种硬币组合能组成总金额，返回 -1。

你可以认为每种硬币的数量是无限的。

示例1：

输入：coins = [1, 2, 5], amount = 11
输出：3 
解释：11 = 5 + 5 + 1

示例2：

输入：coins = [2], amount = 3
输出：-1

示例3：

输入：coins = [1], amount = 0
输出：0

示例4：

输入：coins = [1], amount = 1
输出：1

示例5：

输入：coins = [1], amount = 2
输出：2

提示：

  • $1 <= coins.length <= 12$
  • $1 <= coins[i] <= 10^{31}-1$
  • $0 <= amount <= 10^4$

题解

func coinChange(coins []int, amount int) int {
    dp := make([]int, amount+1) //记录每个金额所需要的最小金币数
    for i,_ := range dp{
        dp[i] = amount+1
    }
    dp[0] = 0 
    for i:= 1; i<= amount; i++{
        for j:=0; j<len(coins);j++{
            if coins[j] <= i{
                dp[i] = min(dp[i], dp[i - coins[j]] + 1)
            }
        }
    }
    if dp[amount] > amount{
        return -1
    }
    return dp[amount]
}

func min(a,b int) int{
    if a > b{
        return b
    }
    return a
}

思路

我们假设$f(i)$等于金额i的时候所需最小的硬币数

我们从上往下推导

那么 $f(i)=min(f(i-c_j)+1)$

其中$c_j$代表的是第$j$枚硬币的面值

根据公式开始写代码

func coinChange(coins []int, amount int) int {
    dp := make([]int, amount+1) //记录每个金额所需要的最小金币数
    for i,_ := range dp{
        dp[i] = amount+1 //因为求最小值，先假设都是无解
    }
    dp[0] = 0 //金额0的时候，最近硬币数也是0
    for i:= 1; i<= amount; i++{ //开始遍历每个金额数
        for j:=0; j<len(coins);j++{
            //f(i-c) 所以使用的硬币数一定要小于当前金额
            if coins[j] <= i{ 
                dp[i] = min(dp[i], dp[i - coins[j]] + 1)
            }
        }
    }
    //如果这个金额的硬币数大于金额，说明无法完成
    if dp[amount] > amount{ 
        return -1
    }
    return dp[amount]
}

func min(a,b int) int{
    if a > b{
        return b
    }
    return a
}
